Q2. "Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb Output: a=2 b=2 c=2 d=1 e=2 f=1" > 

Ans.   string= "abcadeecfb"
          arr = string.split("")
          function countAlphabet(arr){
              let map = new Map();
              for(item of arr){
                  if(map.has(item)){
                      map.set(item , map.get(item)+1)
                  }
                  else{
                      map.set(item , 1)
                  }
              }
              return map
          }

      let count = countAlphabet(arr);
      count.forEach((alphabet,count)=>{
          console.log(`${count} : ${alphabet}`)
      })
